% ============================
% 1. SIN RESIDUO – UNA VARIABLE
% ============================

\begin{ejercicio}
Calcule $\dfrac{x^2+5x+6}{x+2}$.
\end{ejercicio}
\muestraSolucion{$x+3$}

\begin{ejercicio}
Calcule $\dfrac{x^3+4x^2+5x+2}{x+1}$.
\end{ejercicio}
\muestraSolucion{$x^2+3x+2$}

\begin{ejercicio}
Calcule $\dfrac{x^3-8}{x-2}$.
\end{ejercicio}
\muestraSolucion{$x^2+2x+4$}

\begin{ejercicio}
Calcule $\dfrac{2x^3+3x^2-2x-3}{2x+3}$.
\end{ejercicio}
\muestraSolucion{$x^2-1$}

\begin{ejercicio}
Calcule $\dfrac{x^4-1}{x^2-1}$.
\end{ejercicio}
\muestraSolucion{$x^2+1$}

% ============================
% 2. SIN RESIDUO – DOS VARIABLES
% ============================

\begin{ejercicio}
Calcule $\dfrac{x^2+2xy+y^2}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x+y$}

\begin{ejercicio}
Calcule $\dfrac{x^3+3x^2y+3xy^2+y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x^2+2xy+y^2$}

\begin{ejercicio}
Calcule $\dfrac{x^2y+3xy^2+2y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$xy+2y^2$}

\begin{ejercicio}
Calcule $\dfrac{x^3+3x^2y+3xy^2+y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x^2+2xy+y^2$}

\begin{ejercicio}
Calcule $\dfrac{x^2y^2+2xy^3+y^4}{xy+y^2}$.
\end{ejercicio}
\muestraSolucion{$xy+y^2$}

% 3. CON RESIDUO – UNA VARIABLE
% ============================

\begin{ejercicio}
Calcule $\dfrac{x^3+2x^2+3}{x+1}$.
\end{ejercicio}
\muestraSolucion{$x^2+x-1$ con residuo $4$}

\begin{ejercicio}
Calcule $\dfrac{x^2+3x+5}{x+2}$.
\end{ejercicio}
\muestraSolucion{$x+1$ con residuo $3$}

\begin{ejercicio}
Calcule $\dfrac{x^4-2x^2+3}{x^2-1}$.
\end{ejercicio}
\muestraSolucion{$x^2-1$ con residuo $2$}

\begin{ejercicio}
Calcule $\dfrac{2x^3-3x+7}{x-2}$.
\end{ejercicio}
\muestraSolucion{$2x^2+4x+5$ con residuo $17$}

\begin{ejercicio}
Calcule $\dfrac{x^3+5}{x+2}$.
\end{ejercicio}
\muestraSolucion{$x^2-2x+4$ con residuo $-3$}

% ============================
% 4. CON RESIDUO – DOS VARIABLES
% ============================

\begin{ejercicio}
Calcule $\dfrac{x^2+xy+y^2}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x$ con residuo $y^2$}

\begin{ejercicio}
Calcule $\dfrac{x^3+2x^2y+y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x^2+xy-y$ con residuo $y^2+y^3$}

\begin{ejercicio}
Calcule $\dfrac{x^2y+2xy^2+3y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$xy+y^2$ con residuo $2y^3$}

\begin{ejercicio}
Calcule $\dfrac{x^2+2xy}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x+y$ con residuo $-y^2$}

\begin{ejercicio}
Calcule $\dfrac{x^3+3xy^2+2y^3}{x+y}$.
\end{ejercicio}
\muestraSolucion{$x^2-xy+4y^2$ con residuo $-2y^3$}
