\begin{ejemplos}
  \item \(\sqrt{-4}\cdot\sqrt{-9}=(2\ii)(3\ii)=6\ii^2=-6\).
  \item \((3\ii+5\sqrt{2}\,\ii)(2\ii-2\sqrt{2}\,\ii)=
        \ii^2\,(3+5\sqrt{2})(2-2\sqrt{2})= -8.25\) y su valor absoluto \(8.25\).
\end{ejemplos}
